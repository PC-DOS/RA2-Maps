                                has_war_with = ROOT
                            }
                        }
                    }
                }
                ROOT = {
                    # Current scope is the winner, target and its subjects are the losers
                    start_peace_conference  = {

                                is_in_faction = yes

                                    has_war_with = PREV
                                }
                            }
                        }
                    }
                    PREV = {

                        }
                    }
